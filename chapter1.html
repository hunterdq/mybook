添加一个帧结构处理类：MyFrame.cs

public class MyFrame

{
  public byte[] msgHead = new byte[] { 0x14, 0x6F };
  public byte[] msgTail = new byte[] { 0xEB, 0x90 };
  public string msgBody;

  public byte[] GetBodyLength()
  {
    return BitConverter.GetBytes(GetBytes().Length);
  }

  public byte[] GetBytes()
  {
    return Encoding.Default.GetBytes(msgBody);
  }

  public void SetBody(byte[] data)
  {
    msgBody = Encoding.Default.GetString(data);
  }
}
